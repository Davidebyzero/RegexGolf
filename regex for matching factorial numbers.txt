(?# Match factorial numbers in the domain ^x*$ )
^
(?=
	(?*(x(x*))(x*$))                    (?# \1 = conjectured for which \1! == tail; \2 = \1-1; \3 = tool to compare against \1 )
	(?!
		(x+)(?=\3)                      (?# \4 = any number <= \1 )
		(?!\4*$)                        (?# there is no \4 that doesn't divide tail )
	)
	(
		(?=
			(?=(x(x(x*?)))\6+$)         (?# \6 = smallest prime factor; \7 = \6-1; \8 = \7-1 )
			(?*(x(x+)))                 (?# \9 = largest prime power factor corresponding to \6; \10 = \9-1 )
			\9*(?=\9$)
			(?# Assert that there is no prime \12 that divides \9 and isn't equal to \6, and that it's also not true that \9 is prime and isn't equal to \6)
			(?!
				((xx+)\12*(?=\12$))?
				(?!(xx+)\13+$|\6$)
			)
		)
		(?=
			.*?(?!\1.)
			(                           (?# \14 = largest power of \6 less than or greater than \1 )
				(                       (?# Assert that tail is a power of \6 )
					(?=                 (?# Divide: \16 = tail / \6, with \17 = \16-1 )
						(x(x*))
						(?=\16*$)       (?# Testing for divisibility by \7 here isn't necessary, as tail is a power of \6 )
						(\7\17+$)       (?# \18 = tool to make the tail equal \16 )
					)\18
				)*x$
			)
		)
		(?=
			.*(?=\9\14$)                                    (?# Seed the loop )
			(
				(?# The first thing we need to do every iteration is decode the current number back into two prime powers,
					\20 and \24, where \20 is the smaller one, if they are different )
				(?=
					(x(x*?))                                (?# \20 = smaller prime power; \21 = \20-1 )
					(?!((x+)(?=\23+$)x+)\22*(?!\23+$)\22$)  (?# Assert that tail is a prime power)
					(\6+|x)$                                (?# \24 = larger prime power )
				)
				(?=
					.*?(?!\1.)                              (?# discard the remainder )
					(?=\20*$)                               (?# discard the remainder )
					(x(x*))                                 (?# \25 = floor{\1 / \20}; \26 = \25-1 )
					(?=\25*$)
					(?=\21+$)
					\21\26*$
				)
				(?# construct \6 ^ \25 by taking logarithm in base \6 and matching it against \25 )
				(?=
					.*
					(?=
						(                                   (?# Assert that tail is a power of \6 )
							(?=                             (?# Divide: \28 = tail / \6, with \29 = \28-1 )
								(x(x*))
								(?=\28*$)                   (?# Testing for divisibility by \7 here isn't necessary, as tail is a power of \6 )
								(\7\29+$)                   (?# \30 = tool to make the tail equal \28 )
							)\30
						)*x$
					)
					(?=x(x*))                               (?# tail will be \6 ^ \25; \31 = tail-1 )
					(?=(x*)\32(x?))                         (?# \32 = floor{tail / 2}, \33\32 = ceil{tail / 2}, \33 = tail % 2 )
					(?=
						(                                       (?# \34 = logarithm in base \6 of tail )
							x                                   (?# seed the loop )
							(?=
								(
									(?=
										(x*)\33(?=\32)          (?# tail = \32;  \36 = tool for decoding current state )
										\36                     (?# tail = current encoded state )
										(x*?)                   (?# \37 = iteration count )
										(?=
											(                   (?# Assert that tail is a power of \6 )
												(?=             (?# Divide: \39 = tail / \6, with \40 = \39-1 )
													(x(x*))
													(?=\39*$)   (?# Testing for divisibility by \7 here isn't necessary, as tail is a power of \6 )
													(\7\40+$)   (?# \41 = tool to make the tail equal \39 )
												)\41
											)*x$
										)
										(x(x*))                 (?# \42 = tail = A ^ {iteration count + 1}; \43 = \42-1 )
									)
									(?# encode state for next loop iteration, so that \37 will be incremented and \42 will be doubled )
									(?=
										(?# commence multiplication of \42 * \7; since we don't know which
											of \7 and \42 is larger/smaller, do two symmetric tests )
										.*
										(?=\42*$)
										(?= \7*$)
										(?=\42\8+$)
										(  \7\43+$)             (?# \44 = \42 * \7 )
									)
									x\44                        (?# tail -= \42 * \7 + 1 )
								)+
							)
							x\37                                (?# Return logarithm in base A of B )
						|                                       (?# The main algorithm can't handle B<=A, so handle that below )
							x?\B                                (?# If B>1 return a value of 1, otherwise return a value of 0 )
						)
					)
					.*(?=\34$)\25$            (?# Assert that \34, the logarithm, equals \25 )
				)
				(?=
					.*(?=\24$)
					(x(x*))                   (?# \45 = \24 / {\31+1 == \6^\25}; \46 = \45-1 )
					(?=\45*$)                 (?# Testing for divisibility by \31 here isn't necessary, as dividend and divisor are both powers of \6 )
					\31\46*$
				)
				(?=
					.*(?=\20$)
					(x(x*))                   (?# \47 = \20 / \6; \48 = \47-1 )
					(?=\47*$)                 (?# Testing for divisibility by \7 here isn't necessary, as tail is a power of \6 )
					\7\48*$
				)
				.*(?=\45\47$)
			)*
			xx$
		)
		(?=
			(x(x*))                           (?# \49 = tail / \9; \50 = \47-1 )
			(?=\49*$)
			(?=\10+$)
			(\10\50*$)                         (?# \51 = tool for making tail equal \49 )
		)\51
	)*
	(?!\1.|(xx+)\52+$)                        (?# Assert that the remaining factor is <= \1 and is prime )
)
\1
